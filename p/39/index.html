# 开始设置代理
$ sudo systemctl edit snapd
加上：
[Service]
Environment="http_proxy=http://127.0.0.1:port"
Environment="https_proxy=http://127.0.0.1:port"

#保存退出。
$ sudo systemctl daemon-reload
$ sudo systemctl restart snapd
